    def _key(self, f, args):
        """Key generator for function *f* with positional arguments *args*."""
        f_args = inspect.getargspec(f).args

        if len(f_args) > 0:
            idx = 0
            if f_args[0] in ('cls', 'self'):
                idx = 1

            argparts = [self.KEY_BASE, f.__name__]
            for arg in args[idx:]:
                argparts.append(unicode(arg))

        key = u':'.join(argparts)
        return key

    def __call__(self, f):
        """Decorate *f* function to enable caching it.

        Cache key will be generated from ``self.KEY_BASE``, function name and
        values of positional args passed to the function."""
        @wraps(f)
        def wrapper(*args, **kwargs):
            key = self._key(f, args)

            data = self.load(key, f_args=args, f_kwargs=kwargs)
            if data:
                data = self.after_load(data, f_args=args, f_kwargs=kwargs)
            else:
                data = f(*args, **kwargs)

                cached = self.before_save(data, f_args=args, f_kwargs=kwargs)
                self.save(key, cached, f_args=args, f_kwargs=kwargs)

                if self._ttl:
                    self.connection.expire(key, self._ttl)

            return data

        return wrapper
